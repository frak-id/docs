# NexusConfigProvider

React context provider for the config of Nexus Wallet SDK.

## Import

```tsx
import { NexusConfigProvider } from "@frak-labs/nexus-sdk/react";
```

## Usage

:::code-group
```tsx [App.tsx]
import { NexusConfigProvider } from "@frak-labs/nexus-sdk/react"; // [!code focus]
import { config } from './config'; // [!code focus]

function App() {
  return (
    <NexusConfigProvider config={config}> // [!code focus]
        {/** ... */}
    </NexusConfigProvider> // [!code focus]
  )
}
```

:::

```tsx
import type {
```
