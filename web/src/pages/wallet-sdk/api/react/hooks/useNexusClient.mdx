# useNexusClient

Hook that returns the Nexus iframe client.

## Import

```tsx
import { useNexusClient } from "@frak-labs/nexus-sdk/react";
```

## Usage

```tsx [index.tsx]
import { useNexusClient } from "@frak-labs/nexus-sdk/react";
import type { WalletStatusReturnType } from "@frak-labs/nexus-sdk/core";

function App() {
  const client = useNexusClient();

  return client.listenerRequest(
        {
            method: "frak_listenToWalletStatus",
        },
        (walletStatus: WalletStatusReturnType) => {
            console.log("Wallet status changed", { walletStatus });
        }
    );
}
```

## Return type

```tsx
import type { NexusClient } from "@frak-labs/nexus-sdk/core";
```

### config

[`NexusWalletSdkConfig`](/wallet-sdk/api/react/hooks/useNexusConfig#return-type)

Nexus Wallet SDK configuration.

### waitForConnection

`Promise<boolean>{:ts}`

Wait for the connection to be established.
