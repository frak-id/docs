# NexusIFrameClientProvider

React context provider for the client of Nexus Wallet SDK.

## Import

```tsx
import { NexusIFrameClientProvider } from "@frak-labs/nexus-sdk/react";
```

## Usage

:::code-group
```tsx [app.tsx]
import {
  NexusConfigProvider,
  NexusIFrameClientProvider  // [!code focus]
} from "@frak-labs/nexus-sdk/react";
import { config } from './config';

function App() {
  return (
    <NexusConfigProvider config={config}>
      <NexusIFrameClientProvider> // [!code focus]
        {/** ... */}
      </NexusIFrameClientProvider> // [!code focus]
    </NexusConfigProvider>
  )
}
```
