# useWalletStatus

Hooks used to listen to the current wallet status.

## Import

```tsx
import { useWalletStatus } from "@frak-labs/nexus-sdk/react";
```

## Usage

```tsx [index.tsx]
import { useWalletStatus } from "@frak-labs/nexus-sdk/react";
import { formatEther } from "viem";

function App() {
  const { data: walletStatus } = useWalletStatus();

  return (
    <div>
      {walletStatus?.key === "connected" && (
          <p>User connected with wallet {walletStatus.wallet}</p>
      )}
      {walletStatus?.key === "not-connected" && <p>User not connected</p>}
    </div>
  );
}
```

## Return type

```tsx
import type {
  WalletStatusQueryReturnType
} from "@frak-labs/nexus-sdk/react";
```

### data

- Waiting for the response

`{ key: "waiting-response"; }`

- When the wallet is connected

`{ key: "connected"; wallet: Address; }`

- When the wallet is not connected

`{ key: "not-connected"; }`

The current wallet status.
